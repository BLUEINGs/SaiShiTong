/ 核心配置类：存储赛程安排的基本参数
class ScheduleConfig {
    LocalTime dayStart = LocalTime.of(8, 0);  // 每日比赛开始时间
    LocalTime dayEnd = LocalTime.of(18, 0);   // 每日比赛结束时间
    int trackCount = 8;                      // 可用跑道数量
    int fieldCount = 2;                      // 田赛场地数量
}

// 比赛项目类：表示一个具体的比赛项目
class SportEvent implements Comparable<SportEvent> {
//    Comparable的中文意思就是可被排序的，代表本身支持排序功能。只要我们的类实现了这个接口，那么这个类的对象就会自动拥有了可被排序的能力。而且这个排序被称为类的自然顺序。
//    这个类的对象的列表可以被Collections.sort和Arrays.sort来执行排序。同时这个类的实例具备作为sorted map的key和sorted set的元素的资格。
//    假如a和b都是实现了Comparable接口的类C的实例，那么只有当a.compareTo(b)的结果与a.equals(b)的结果一致时，才称类C的自然顺序与equals一致。
//    强烈建议将类的自然顺序和equals的结果保持一致，因为如果不一致的话，由该类对象为键的sorted map和由该类对象为元素的sorted set的行为将会变得很怪异。
//    该接口只有一个抽象方法compareTo，这个方法主要就是为了定义我们的类所要排序的方式。compareTo方法用于比较当前元素a与指定元素b，结果为int值，
//    如果a > b，int>0；如果a=b，int=0；如果a<b，int<0。
    String name;            // 项目名称（如"男子100米预赛"）
    EventType type;         // 项目类型（短跑、田赛等）
    int groupCount;         // 需要分多少组
    int participants;       // 总参赛人数
    Set<String> schools = new HashSet<>(); // 涉及学校
//    Java集合框架中的Set接口是Collection接口的一个子接口，它表示一个无序且不包含重复元素的集合。
//    Set接口的主要目的是确保集合中的元素唯一性，即集合中不允许出现重复的元素。
//    它继承自Collection接口，并且有两个主要的实现类：HashSet和TreeSet。
    Set<String> classes = new HashSet<>(); // 涉及班级
//    HashSet是Set接口最常用的实现类之一。
//    它使用哈希表来存储元素，提供快速的添加、删除和查找操作。
//    HashSet不保证元素的顺序，并且允许包含一个null元素。
// 说明:HashSet
//1. 在执行add方法后，会返回一个boolean值
//2. 如果添加成功，返回 true, 否则返回false
//3. 可以通过 remove 指定删除哪个对象


    // 实现比较逻辑：参赛人数多的优先安排
    @Override
    public int compareTo(SportEvent o) {
        return o.participants - this.participants; // 降序排列,谁参赛者多谁在前面
    }
//    compareTo 方法用于排序，参赛人数多的项目优先安排。
//    schools 和 classes 用于冲突检测。
}

// 项目类型枚举：定义不同类型的比赛项目
enum EventType {
    SPRINT,     // 短跑
    MID_DISTANCE, // 中距离
    LONG_DISTANCE,// 长跑
    RELAY,      // 接力
    FIELD       // 田赛
}
